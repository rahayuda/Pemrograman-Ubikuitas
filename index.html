const handleMotion = (event) => { document.getElementById("Accelerometer_gx").innerHTML = event.accelerationIncludingGravity.x.toFixed(2); document.getElementById("Accelerometer_gy").innerHTML = event.accelerationIncludingGravity.y.toFixed(2); document.getElementById("Accelerometer_gz").innerHTML = event.accelerationIncludingGravity.z.toFixed(2); document.getElementById("Accelerometer_x").innerHTML = event.acceleration.x.toFixed(2); document.getElementById("Accelerometer_y").innerHTML = event.acceleration.y.toFixed(2); document.getElementById("Accelerometer_z").innerHTML = event.acceleration.z.toFixed(2); document.getElementById("Gyroscope_z").innerHTML = event.rotationRate.alpha.toFixed(2); document.getElementById("Gyroscope_x").innerHTML = event.rotationRate.beta.toFixed(2); document.getElementById("Gyroscope_y").innerHTML = event.rotationRate.gamma.toFixed(2); }; document.addEventListener("devicemotion", handleMotion); const handleOrientation = (event) => { document.getElementById("Orientation_a").innerHTML = event.alpha.toFixed(2); document.getElementById("Orientation_b").innerHTML = event.beta.toFixed(2); document.getElementById("Orientation_g").innerHTML = event.gamma.toFixed(2); }; document.addEventListener("deviceorientation", handleOrientation);
